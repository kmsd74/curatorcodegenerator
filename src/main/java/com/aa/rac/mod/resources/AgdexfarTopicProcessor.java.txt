package com.aa.rac.mod.resources;

import com.aa.rac.mod.domain.BaseService;
import com.aa.rac.mod.orm.dao.agdexfarrepl.AgdexfarRepl;
import com.aa.rac.mod.repository.eventhub.Agdexfar;
import java.util.function.Consumer;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.beans.factory.annotation.Qualifier;
import org.springframework.beans.factory.annotation.Value;
import org.springframework.context.annotation.Bean;
import org.springframework.messaging.Message;
import org.springframework.stereotype.Component;

/** Agdexfar topic processor. */
@Component
public class AgdexfarTopicProcessor 
    extends AbstractTopicProcessor {

  @Autowired 
  BaseService<Agdexfar, AgdexfarRepl> service;

  @Value(#TODO) 
  String topicName;

  /**
   * Recieve Agdexfar consumer.
   *
   * @return the consumer
   */
  @Bean
  public Consumer<Message<String>> receiveAgdexfar() {
    return consume(topicName);
  }

  @Override
  public BaseService getService() {
    return service;
  }
}
