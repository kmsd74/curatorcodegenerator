package com.aa.rac.mod.resources.kafka;

import com.aa.rac.mod.orm.dao.refundedrepl.RefundedRepl;
import com.aa.rac.mod.repository.eventhub.Refunded;
import com.aa.rac.mod.domain.BaseService;
import io.github.resilience4j.circuitbreaker.annotation.CircuitBreaker;
import java.util.concurrent.ExecutionException;
import org.apache.kafka.clients.consumer.ConsumerRecords;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.beans.factory.annotation.Qualifier;
import org.springframework.boot.autoconfigure.condition.ConditionalOnProperty;
import org.springframework.context.annotation.Profile;
import org.springframework.kafka.annotation.KafkaListener;
import org.springframework.kafka.support.Acknowledgment;
import org.springframework.stereotype.Component;

/** Refunded topic processor. */
@Component
@Profile("!JunitTest")
@ConditionalOnProperty(name = "spring.kafka.topics.recieve.refunded.enabled", havingValue = "true")
public class RefundedTopicProcessor 
    extends AbstractTopicProcessor {

  @Autowired
  public RefundedTopicProcessor(
      @Qualifier("RefundedReplServiceImpl")
      BaseService<Refunded, RefundedRepl> baseService) {
    super(baseService);
  }

  @KafkaListener(
      id = "${spring.kafka.topics.recieve.refunded.destination}",
      topics = "#{'${spring.kafka.topics.recieve.refunded.destination}'.split(',')}",
      concurrency = "${CONSUMER_CONCURRENCY_COUNT}",
      clientIdPrefix = "refunded",
      batch = "true",
      properties = {
        "auto.offset.reset:earliest",
        "reconnect.backoff.max.ms:5000",
        "reconnect.backoff.ms:1000",
        "retry.backoff.ms:5000"
      })
  @CircuitBreaker(name = "${spring.kafka.topics.recieve.refunded.destination}")
  public void refunded(ConsumerRecords<String, String> payload, Acknowledgment acknowledgment)
      throws InterruptedException, ExecutionException {
    processItems(payload);
    acknowledgment.acknowledge();
  }}
