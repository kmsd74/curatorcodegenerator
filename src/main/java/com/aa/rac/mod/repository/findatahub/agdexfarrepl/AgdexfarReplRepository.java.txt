package com.aa.rac.mod.repository.findatahub.agdexfarrepl;

import com.aa.rac.mod.orm.dao.agdexfarrepl.AgdexfarRepl;
import java.util.Optional;
import org.springframework.data.jpa.repository.JpaRepository;
import org.springframework.data.jpa.repository.Query;
import org.springframework.stereotype.Repository;

/** AgdexfarRepl repository. */
@Repository
public interface AgdexfarReplRepository extends JpaRepository<AgdexfarRepl, String> { 

  @Query(value = "SELECT * FROM replicated.agdexfar"
                + " WHERE ticket_uuid=?1", 
                  nativeQuery = true)
  Optional<AgdexfarRepl> findByTicketUuid(String ticketUuid);
}
