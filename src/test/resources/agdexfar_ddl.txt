CREATE TABLE IF NOT EXISTS replicated.agdexfar
(
    TICKET_UUID                   CHAR(64) NOT NULL, 
    XTD_TICKET_NO                 VARCHAR(14) NOT NULL, 
    XTD_FARE_CALC                 VARCHAR(2000), 
    XTD_CREATED_BY                VARCHAR(8) NOT NULL, 
    XTD_CREATED_TIME              TIMESTAMP WITHOUT TIME ZONE NOT NULL, 
    XTD_MODIFIED_BY               VARCHAR(8) NOT NULL, 
    XTD_MODIFIED_TIME             TIMESTAMP WITHOUT TIME ZONE NOT NULL, 
    XTD_PARTITION_NBR             SMALLINT NOT NULL, 
    TICKET_CREATE_TS              TIMESTAMP WITHOUT TIME ZONE NOT NULL, 
    src_deleted_indicator         BOOLEAN NOT NULL, 
    deleted_indicator             BOOLEAN NOT NULL, 
    dml_flg                       VARCHAR(3) NOT NULL, 
    eventhub_timestamp            TIMESTAMP WITHOUT TIME ZONE NOT NULL, 
    system_modified_timestamp     TIMESTAMP WITHOUT TIME ZONE NOT NULL, 
    created_by                    VARCHAR(100) NOT NULL, 
    modified_by                   VARCHAR(100) NOT NULL, 
    version                       BIGINT NOT NULL, 
    CONSTRAINT replicated_agdexfar_ticket_uuid_pkey PRIMARY KEY (TICKET_UUID)
);

COMMENT ON TABLE replicated.agdexfar IS 'UUID values are derived using SHA256 Hash';
COMMENT ON COLUMN replicated.agdexfar.TICKET_UUID IS 'TICKET_UUID is derived using SHA256 Hash';
